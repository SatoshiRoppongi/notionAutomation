lets(
	textDate, formatDate(prop("実行年月日"),"YYYY年MM月DD日"),
	year, substring(textDate, 0,4),
	month, substring(textDate, 5,7),
	day, substring(textDate, 8,10),
	mydate, parseDate(year + "-" + month + "-" + day),
if(dateBetween(mydate, now(), "days") < 0, "完了", if(dateBetween(mydate, now(), "days") == 0, "本日実行", "未実行"))
)
